  <script>
    // Nav highlight
    const navItems = document.querySelectorAll(".navp");
    const highlight = document.querySelector(".nav-highlight");

    navItems.forEach(item => {
      item.addEventListener("mouseenter", () => {
        const rect = item.getBoundingClientRect();
        const parentRect = item.parentElement.getBoundingClientRect();
        highlight.style.left = (rect.left - parentRect.left) + "px";
        highlight.style.width = rect.width + "px";
      });
    });

    // Popup logic
    let baseprice = 0;

    function popupform(foodname, img, price) {
      baseprice = price;
      const popup = document.getElementById("popup");
      const overlay = document.getElementById("overlay");

      popup.style.display = "block";
      overlay.classList.add("active");
      document.getElementById("popupimg").src = img;

      const qtyVal = document.getElementById("qtyval");
      const qtyRange = document.getElementById("qtyRange");
      qtyRange.value = 1;
      qtyVal.innerText = 1;

      const priceVal = document.getElementById("priceval");
      priceVal.innerText = baseprice;

      qtyRange.oninput = () => {
        qtyVal.innerText = qtyRange.value;
        priceVal.innerText = qtyRange.value * baseprice;
      };

      // Trigger animation next frame
      requestAnimationFrame(() => popup.classList.add("animatepopup"));
    }

    function closePopup() {
      const popup = document.getElementById("popup");
      const overlay = document.getElementById("overlay");
      popup.classList.remove("animatepopup");
      overlay.classList.remove("active");
      setTimeout(() => popup.style.display = "none", 200);
    }

    function sucessespage() {
      window.location.href = "sucess.html";
    }

    function samepage() {
      window.location.href = "veg.html";
    }
